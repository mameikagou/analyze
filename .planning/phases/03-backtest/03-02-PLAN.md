```python
def _build_target_weights(self, scores: pd.DataFrame) -> pd.DataFrame:
    # 1. Generate rebalance dates from scores.index using resample(rebalance_freq).last()
    rebalance_dates = scores.resample(self.config.rebalance_freq).last().index
    rebalance_dates = rebalance_dates.intersection(scores.index)
    # 2. Initialize weights matrix with all NaN (vbt ignores NaN = hold current)
    weights = pd.DataFrame(np.nan, index=scores.index, columns=scores.columns)
    # 3. For each rebalance date:
    #    - valid_scores = row[row > -np.inf] (exclude signal-filtered)
    #    - top_funds = valid_scores.nlargest(self.config.top_n)
    #    - If equal: w = 1.0 / len(top_funds), assign to top funds, 0.0 to others
    #    - If score: min-max normalize shifted scores, assign normalized weights, 0.0 to others
    #    - If no valid funds: weights.loc[dt] = 0.0 (all cash)
    return weights
```
